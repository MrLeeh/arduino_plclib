SR sr_latch;

void setup() {
    pinMode(0, INPUT_PULLUP);
    pinMode(1, INPUT_PULLUP);
    Serial.begin(9600);
}

void loop() {
    boolean x0 = !digitalRead(0); //set switch
    boolean x1 = !digitalRead(1); //reset switch

    sr_latch.process(x0, x1);
    rtrig.process(sr_latch.Q1);
    ftrig.process(sr_latch.Q1);

    if (rtrig.Q) {
        Serial.println("On");
    }
    if (ftrig.Q) {
