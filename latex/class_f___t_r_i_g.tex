\subsection{Detailed Description}
Detect a falling edge. 

Code sample\+: \begin{DoxyVerb}#include "plc_standardlib.h"
#define X0 2

F_TRIG trig;
void setup() {
    pinMode(X0, INPUT_PULLUP);
    Serial.begin(9600);
}

void loop() {
    boolean x0 = !digitalRead(X0);
    trig.process(x0);
    if (trig.Q) {
        Serial.println("This was a falling edge.");
    }
}\end{DoxyVerb}
